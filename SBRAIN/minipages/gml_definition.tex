\begin{definizione}[Multilevel graph]
    A \textbf{multilevel graph} $M$ is a couple $(G, \Gamma)$ where:
        \begin{itemize}
            \item $G = (V, E)$ is a graph;
            \item $\Gamma = \langle \, f_{C_1}, f_{C_2}, .., f_{C_k} \rangle$ is a sequence of contraction functions.
        \end{itemize}
    \end{definizione}
